javascript
Copy code
// Example of using Redux in a React SPA
const counterReducer = (state = 0, action) => {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
    default:
      return state;
  }
};

// Store setup and usage
const store = createStore(counterReducer);
