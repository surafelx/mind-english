# Blog Carousel

<Steps>

### HTML Structure

To start, modify your HTML structure to include a container for the carousel and each blog post wrapped in a `<div>`:

```html
<!-- Main content section -->
<section class="carousel-wrapper">
  <section class="carousel-container">
    <!-- Article 1 -->
    <div class="carousel-slide">
      <!-- Article 1 content -->
    </div>

    <!-- Article 2 -->
    <div class="carousel-slide">
      <!-- Article 2 content -->
    </div>

    <!-- Add more articles as needed -->
  </section>
</section>
</section>
```

### CSS Styling

Next, add some CSS styles to create the layout for the carousel:

```css
.carousel-wrapper {
  overflow: hidden;
}
.carousel-container {
  display: flex;
  width: 100%;
  transition: transform 0.5s ease-in-out;
}

.carousel-slide {
  flex: 0 0 100%;
  box-sizing: border-box;
  padding: 15px;
}

.carousel-nav {
  display: flex;
  justify-content: space-between;
  margin-top: 10px;
}

.carousel-nav button {
  cursor: pointer;
  background-color: black;
  color: beige;
  border: none;
  padding: 10px 15px;
  border-radius: 5px;
}
```

### JavaScript for Carousel Functionality

Now, let's add JavaScript to make the carousel functional. We'll create functions to handle sliding left and right:

```html
<!-- Add this script to the bottom of your body section -->

<script>
  document.addEventListener("DOMContentLoaded", function () {
    var carouselContainer = document.querySelector(".carousel-container");
    var slides = document.querySelectorAll(".carousel-slide");
    var currentIndex = 0;

    function updateCarousel() {
      carouselContainer.style.transform =
        "translateX(" + -currentIndex * 100 + "%)";
    }

    function slideRight() {
      if (currentIndex < slides.length - 1) {
        currentIndex++;
      } else {
        currentIndex = 0;
      }
      updateCarousel();
    }

    function slideLeft() {
      if (currentIndex > 0) {
        currentIndex--;
      } else {
        currentIndex = slides.length - 1;
      }
      updateCarousel();
    }

    // Optional: Add navigation buttons
    var navContainer = document.createElement("div");
    navContainer.className = "carousel-nav";

    var prevButton = document.createElement("button");
    prevButton.textContent = "Previous";
    prevButton.addEventListener("click", slideLeft);

    var nextButton = document.createElement("button");
    nextButton.textContent = "Next";
    nextButton.addEventListener("click", slideRight);

    navContainer.appendChild(prevButton);
    navContainer.appendChild(nextButton);

    document.body.appendChild(navContainer);
  });
</script>
```
